```typescript
/**
 * 快速排序：分治算法
 * 时间复杂度：O(n log n) 平均情况，O(n²) 最坏情况
 * 空间复杂度：O(log n) 递归栈深度
 */
function quickSort(arr: number[], left = 0, right = arr.length - 1): number[] {
  if (left >= right) return arr;

  // 分割
  const pivot = partition(arr, left, right);

  // 递归排序左右两部分
  quickSort(arr, left, pivot - 1);
  quickSort(arr, pivot + 1, right);

  return arr;
}

/**
 * 分割函数：选择枢纽，将数组分为两部分
 */
function partition(arr: number[], left: number, right: number): number {
  const pivot = arr[right];
  let i = left - 1;

  for (let j = left; j < right; j++) {
    if (arr[j] < pivot) {
      i++;
      [arr[i], arr[j]] = [arr[j], arr[i]]; // 交换
    }
  }

  [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
  return i + 1;
}

// 测试
const arr = [64, 34, 25, 12, 22, 11, 90];
console.log(quickSort(arr)); // [11, 12, 22, 25, 34, 64, 90]
```

### 优化后的实现

```typescript
/**
 * 优化版本：三路快速排序
 * 优势：处理重复元素时性能更好
 */
function quickSortOptimized(
  arr: number[],
  left = 0,
  right = arr.length - 1
): number[] {
  if (left >= right) return arr;

  // 随机选择枢纽，避免最坏情况（O(n²)）
  const randomIndex = left + Math.floor(Math.random() * (right - left + 1));
  [arr[randomIndex], arr[right]] = [arr[right], arr[randomIndex]];

  // 三路分割
  const [lt, gt] = partition3Way(arr, left, right);

  quickSortOptimized(arr, left, lt - 1);
  quickSortOptimized(arr, gt + 1, right);

  return arr;
}

/**
 * 三路分割：返回 [小于枢纽的最后位置, 大于枢纽的第一位置]
 * 处理重复元素时，中间部分不需要再排序
 */
function partition3Way(
  arr: number[],
  left: number,
  right: number
): [number, number] {
  const pivot = arr[right];
  let lt = left;        // < pivot 的最后位置
  let gt = right - 1;   // > pivot 的第一位置
  let i = left;

  while (i <= gt) {
    if (arr[i] < pivot) {
      [arr[lt++], arr[i++]] = [arr[i], arr[lt]];
    } else if (arr[i] > pivot) {
      [arr[i], arr[gt--]] = [arr[gt], arr[i]];
    } else {
      i++;
    }
  }

  [arr[gt + 1], arr[right]] = [arr[right], arr[gt + 1]];
  return [lt, gt + 1];
}
```

## 关键学习点

### 1. 快速排序的两个版本

| 版本 | 优点 | 缺点 | 适用 |
|------|------|------|------|
| 基础版 | 简单易懂 | 重复元素时慢 | 学习 |
| 三路版 | 处理重复高效 | 稍复杂 | 生产 |

### 2. 随机化的重要性

```typescript
// 基础版可能退化到 O(n²)
arr = [1, 2, 3, 4, 5]; // 已排序，选最后一个元素作枢纽 = 最坏情况

// 随机化避免这个问题
const randomIndex = left + Math.floor(Math.random() * (right - left + 1));
[arr[randomIndex], arr[right]] = [arr[right], arr[randomIndex]];
```

### 3. 三路分割的价值

当有很多重复元素时：
```typescript
// 基础版：[1, 1, 1, 1, 2, 3] 需要递归排序 [1, 1, 1, 1]
// 三路版：知道中间的 [1, 1, 1, 1] 已排序，不再递归
```
