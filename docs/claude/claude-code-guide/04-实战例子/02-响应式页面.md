// 使用 React.memo 防止不必要的重新渲染
export const ProductCard = React.memo(ProductCardComponent, (prev, next) => {
  return (
    prev.product.id === next.product.id &&
    prev.onAddCart === next.onAddCart
  );
});

// 使用 useCallback 保证回调函数引用不变
const handleAddCart = useCallback(() => {
  onAddCart(product.id);
}, [product.id, onAddCart]);

// 图片懒加载

## 响应式设计要点

### Tailwind 断点系统

```typescript
// 默认（手机优先）
className="text-sm" // 所有屏幕 14px

// 平板以上
className="text-sm md:text-base" // md: 16px

// 桌面以上
className="text-sm md:text-base lg:text-lg" // lg: 18px

// 超大屏
className="text-sm md:text-base lg:text-lg xl:text-xl"
```

### 常见响应式模式

| 模式 | 代码 | 说明 |
|------|------|------|
| 一到多列 | `grid-cols-1 md:grid-cols-2 lg:grid-cols-3` | 单列→双列→三列 |
| 隐藏/显示 | `hidden md:block` | 手机隐藏，平板显示 |
| 字号调整 | `text-sm md:text-base lg:text-lg` | 根据屏幕调整 |
| 间距调整 | `p-2 md:p-4 lg:p-6` | 响应式内边距 |
